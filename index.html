function fetchTextWithCorsFallback(url) { let proxies = [ 'https://cors-anywhere.herokuapp.com/', 'https://thingproxy.freeboard.io/fetch/', 'https://api.allorigins.win/get?url=', ]; let fetchPromises = proxies.map(proxy => { return fetch(proxy + encodeURIComponent(url)) .then(response => { if (!response.ok) throw new Error('Network response was not ok'); return response.text(); }); }); return Promise.any(fetchPromises) .catch(error => { console.error('All proxy attempts failed:', error); throw new Error('All CORS proxy attempts failed.'); }); } function checkCorsSupport() { return new Promise((resolve) => { const xhr = new XMLHttpRequest(); xhr.open('GET', '/test-cors', true); xhr.onload = () => resolve(xhr.status === 200); xhr.onerror = () => resolve(false); xhr.send(); }); } function handleCorsRequest(url) { return checkCorsSupport().then((supported) => { if (supported) { return fetch(url).then(response => response.text()); } else { return fetchTextWithCorsFallback(url); } }); }
